float t;

int r;

 

 

void setup() {

  size(480, 480);

  r=120;

}

 

void draw() {

 

  float x=0;

  float y=0;

 

 

background(40, 255, 0);

 

 

//  moving

  fill(255);

  x=r*sin(t);

  y=r*cos(t);

 

//  cord

  stroke(255);

  line(120, 240, x+240, y+240);

  line(360, 240, x+240, y+240);

  if (y+240>=240) {

    t+=0.15;

  }

  else {

    t+=0.05;

  }

 

 

//  diabolo

  noStroke();

  fill(0);

  ellipse(x+240, y+240, 20, 20);

  fill(255, 150, 0);

  ellipse(x+240, y+240, 60, 60);

  fill(200);

  ellipse(x+240, y+240, 20, 20);

}
