float rx = 0.0;

float ry = 0.0;

float rz = 0.0;

float z;

  PImage img;

void setup(){

 img=loadImage("ball.jpg");

 size(displayWidth,displayHeight,P3D);

  smooth();

  frameRate(50); 

}

void draw(){

   background(200);

 image(img,0,0,displayWidth,displayHeight);

 z += 10;

   pushMatrix(); 

  translate(width/2.5,height/2,z);

  rx += 0.5;

  ry += 0.5;

  rotateX(radians(rx));

  rotateY(radians(ry));

  sphere(40);

  popMatrix();

 

  pushMatrix(); 

  translate(width/1.7,height/2,z);

  rx += 1;

  ry += 1;

  rotateX(radians(rx));

  rotateY(radians(ry));

  box(100);

  popMatrix();

  if(z > 800){

    z = 0;

  }

}
